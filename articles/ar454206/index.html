import sys

import numpy as np
from keras.models import load_model
from keras.preprocessing import image
data_path = "../data/"
picture_path = data_path + "{}.jpg"
CLASSES = [1, 0]
def run(model_path):

pic_num = len(pictures_names)

res = []

n = int("0b" + res, 2)

if len(sys.argv) < 2:
