Parallelschaltung mehrerer Röhren

//   / 
int pocet;                            //    
unsigned long time;                   //    

void setup() {
  pinMode(2, INPUT);                  // pin 2     
  attachInterrupt(0, nacti, RISING);  //  
  Serial.begin(9600);                 //       
  Serial.println(" ");                //    
}

void nacti() {
  pocet = pocet++;                    //  int0
}

void loop() {
  pocet = 0;                          //  
  time = millis() + 60000;            //   
  while (time > millis()) {}          //   1 
  if (pocet < 10) Serial.print(" ");  //    
  if (pocet < 100) Serial.print(" ");
  if (pocet < 1000) Serial.print(" ");
  Serial.print(pocet);                //   /
  Serial.print(" ");
  for (int i = 0; i < pocet; i++) {   //  
    Serial.print("#");
  }
  Serial.println(" ");                //  
}
