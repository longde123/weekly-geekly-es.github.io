}

clockwise();
else if (count == countsperrev * 2)
count = 0;
else
anticlockwise();
count++;
}

//delay ¬´motorSpeed¬ª between each pin setting (to determine speed)

{

{
setOutput(i);
delayMicroseconds(motorSpeed);
}
}

setOutput(i);
delayMicroseconds(motorSpeed);

}
