给上釉比自负好


许多人可能仍然不太了解如何使用所描述的小工具。也许最简单的方法是在两种标准情况下进行说明：
  • 我们有一个数字温度计，有必要在显示器上显示其读数。
  • 有一个伺服驱动器，需要设置其操作参数。例如，他的位置。

温度计

#include <intelcd.h>
#include <OneWire.h>
#include <DallasTemperature.h>
 
//     2 Arduino   1-Wire
#define ONE_WIRE_BUS 2
 
/*************************************************************
*     ,                      * 
*      :                      *
*************************************************************/
//  ,   . 
#define FORM_TERMO 0
//  ,    
#define FIELD_TERMO 0
 
 
//   oneWire     oneWire
OneWire oneWire(ONE_WIRE_BUS); 
//   sensor    
DallasTemperature sensor(&oneWire);
//   lcd   LCD 
InteLCD lcd;
 
void setup(void)
{
    lcd.init(); 
    sensor.begin(); 
}
 
void loop(void)
{
    //   
    sensor.requestTemperatures();
 
    //     
    lcd.update(FORM_TERMO, FIELD_TERMO, sensor.getTempCByIndex(0));
    delay(1000);
}
